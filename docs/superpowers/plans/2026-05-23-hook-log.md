# Hook Activity Log Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Record every PreToolUse hook invocation to `.agentfence/events.ndjson` and expose it via `agentfence hook-log` with a formatted terminal display and 30-day stats.

**Architecture:** Single append-only NDJSON file per project in `.agentfence/events.ndjson`. `appendHookLogEntry()` is called from `runHookInputCheck` before every exit. `readHookLog()` reads + lazily prunes on demand. `agentfence hook-log` groups output by day with icons and a 30-day summary footer.

**Tech Stack:** TypeScript ESM, Node.js `fs/promises` (appendFile, readFile, writeFile, mkdir), Vitest, spawnSync for CLI integration tests, chalk for terminal output.

```typescript
import { appendFile, mkdir, readFile, writeFile } from "node:fs/promises";
import path from "node:path";
import type { HookLogEntry, HookLogTier, HookLogOutcome } from "../../types/index.js";
import type { HookTool } from "../scanner/sensitive-paths.js";

export function hookLogPath(root?: string): string {
  return path.join(root ?? process.cwd(), ".agentfence", "events.ndjson");
}

export async function appendHookLogEntry(
  filePath: string,
  tool: HookTool,
  outcome: HookLogOutcome,
  tier?: HookLogTier,
  ruleId?: string,
  root?: string
): Promise<void> {
  try {
    const logPath = hookLogPath(root);
    await mkdir(path.dirname(logPath), { recursive: true });
    const entry: HookLogEntry = {
      ts: new Date().toISOString(),
      tool,
      filePath,
      outcome,
      ...(tier !== undefined ? { tier } : {}),
      ...(ruleId !== undefined ? { ruleId } : {}),
    };
    await appendFile(logPath, JSON.stringify(entry) + "\n");
  } catch {
    // never throw — a logging failure must not block hook invocations
  }
}

export async function readHookLog(
  root?: string,
  opts?: { since?: Date; prune?: boolean }
): Promise<HookLogEntry[]> {
  const logPath = hookLogPath(root);
  let raw: string;
  try {
    raw = await readFile(logPath, "utf8");
  } catch {
    return [];
  }

  const cutoff = new Date();
  cutoff.setDate(cutoff.getDate() - 90);

  const kept: HookLogEntry[] = [];
  let prunedCount = 0;

  for (const line of raw.split("\n")) {
    const trimmed = line.trim();
    if (!trimmed) continue;
    try {
      const entry = JSON.parse(trimmed) as HookLogEntry;
      if (new Date(entry.ts) < cutoff) {
        prunedCount++;
      } else {
        kept.push(entry);
      }
    } catch {
      // skip corrupt lines
    }
  }

  if (opts?.prune && prunedCount > 0) {
    const total = kept.length + prunedCount;
    if (prunedCount / total > 0.1) {
      void writeFile(logPath, kept.map((e) => JSON.stringify(e)).join("\n") + "\n").catch(() => {});
    }
  }

  const since = opts?.since;
  const result = since ? kept.filter((e) => new Date(e.ts) >= since) : kept;
  return result.sort((a, b) => a.ts.localeCompare(b.ts));
}
```

```typescript
async function runHookInputCheck(toolName: HookTool): Promise<void> {
  // Step 1: Read stdin and parse JSON payload
  const chunks: Buffer[] = [];
  for await (const chunk of process.stdin) chunks.push(chunk as Buffer);

  let payload: Record<string, unknown> = {};
  try {
    payload = JSON.parse(Buffer.concat(chunks).toString("utf8")) as Record<string, unknown>;
  } catch {
    // Malformed JSON payload — fail open, no filePath to log
    process.exit(0);
  }

  const toolInput = (payload.tool_input ?? {}) as Record<string, unknown>;
  const filePath = (toolInput.file_path as string | undefined) ?? "";

  // Step 2: Load policy
  const policy = await loadMergedPolicy();

  // Step 3: Check exceptions — if the file+op is explicitly allowed, exit silently
  if (filePath && matchesException(filePath, toolName, policy.exceptions ?? [])) {
    await appendHookLogEntry(filePath, toolName, "exception");
    process.exit(0);
  }

  // Step 4: Sensitive path check with tier-based response
  const pathResult = checkSensitivePath(filePath, toolName);
  if (pathResult) {
    if (pathResult.tier === "advisory") {
      // Advisory: inject context into Claude's next message, then continue to content scan
      console.log(
        JSON.stringify({
          hookSpecificOutput: {
            hookEventName: "PreToolUse",
            additionalContext: pathResult.message,
          },
        })
      );
      // Don't exit here — fall through to content scan; outcome logged at final exit
    } else {
      // high or critical: show ask dialog and exit
      console.log(
        JSON.stringify({
          hookSpecificOutput: {
            hookEventName: "PreToolUse",
            permissionDecision: "ask",
            permissionDecisionReason: pathResult.message,
          },
        })
      );
      await appendHookLogEntry(filePath, toolName, "ask", pathResult.tier, pathResult.ruleId);
      process.exit(0);
    }
  }

  // Step 5: Content scan (Write and Edit only — Read has no content yet)
  let content = "";
  if (toolName === "Write") content = (toolInput.content as string | undefined) ?? "";
  else if (toolName === "Edit") content = (toolInput.new_string as string | undefined) ?? "";

  if (content) {
    const result = scanContent(content, policy);
    const blocking = result.matches.filter(
      (m) => m.severity === "high" || m.severity === "critical"
    );
    if (blocking.length > 0) {
      const reasons = blocking
        .map((m) => `${m.ruleId} (${m.severity}): ${m.match}`)
        .join("; ");
      console.log(
        JSON.stringify({
          hookSpecificOutput: {
            hookEventName: "PreToolUse",
            permissionDecision: "deny",
            permissionDecisionReason: `[agentfence] content policy violation — ${reasons}`,
          },
        })
      );
      await appendHookLogEntry(filePath, toolName, "denied", undefined, blocking[0].ruleId);
      process.exit(0);
    }
  }

  // Step 6: All checks passed
  // If pathResult was advisory, the outcome for this invocation is "advisory"
  if (pathResult?.tier === "advisory") {
    await appendHookLogEntry(filePath, toolName, "advisory", "advisory", pathResult.ruleId);
  } else {
    await appendHookLogEntry(filePath, toolName, "clean");
  }
  process.exit(0);
}
```

```typescript
import chalk from "chalk";
import path from "node:path";
import { writeFile } from "node:fs/promises";
import { readHookLog, hookLogPath } from "../../core/hook-log/index.js";
import type { HookLogEntry } from "../../types/index.js";

interface HookLogOptions {
  days?: string;
  summary?: boolean;
  json?: boolean;
  prune?: boolean;
}

const ICONS: Record<HookLogEntry["outcome"], string> = {
  clean: "✓",
  advisory: "ℹ",
  ask: "⚠",
  denied: "🛡",
  exception: "⚪",
};

function displayPath(filePath: string): string {
  const normalized = filePath.replace(/\\/g, "/");
  const parts = normalized.split("/").filter(Boolean);
  if (parts.length <= 1) return filePath;
  return parts.slice(-2).join("/");
}

function formatTime(ts: string): string {
  const d = new Date(ts);
  const hh = String(d.getHours()).padStart(2, "0");
  const mm = String(d.getMinutes()).padStart(2, "0");
  return `${hh}:${mm}`;
}

function toDateKey(ts: string): string {
  return ts.slice(0, 10); // "YYYY-MM-DD"
}

function dayLabel(dateKey: string, todayKey: string, yesterdayKey: string): string {
  const date = new Date(dateKey + "T12:00:00Z");
  const label = date.toLocaleDateString("en-US", {
    month: "long",
    day: "numeric",
    year: "numeric",
    timeZone: "UTC",
  });
  if (dateKey === todayKey) return `Today  (${dateKey})`;
  if (dateKey === yesterdayKey) return `Yesterday  (${dateKey})`;
  return label;
}

function outcomeLabel(entry: HookLogEntry): string {
  if (entry.outcome === "denied") {
    return chalk.red(`BLOCKED${entry.ruleId ? ` — ${entry.ruleId}` : ""}`);
  }
  if (entry.outcome === "ask") return chalk.yellow("ask dialog shown");
  if (entry.outcome === "advisory") return chalk.blue("warned Claude about secrets");
  if (entry.outcome === "exception") return chalk.dim("bypassed (policy exception)");
  return chalk.dim("clean");
}

function renderSummaryLine(entries: HookLogEntry[]): string {
  const total = entries.length;
  const blocked = entries.filter((e) => e.outcome === "denied").length;
  const asks = entries.filter((e) => e.outcome === "ask").length;
  const advisories = entries.filter((e) => e.outcome === "advisory").length;
  const clean = entries.filter((e) => e.outcome === "clean" || e.outcome === "exception").length;
  return (
    `  ${total} total  ·  ` +
    `${chalk.red(String(blocked))} blocked  ·  ` +
    `${chalk.yellow(String(asks))} asks  ·  ` +
    `${chalk.blue(String(advisories))} advisories  ·  ` +
    `${chalk.dim(String(clean))} clean`
  );
}

export async function hookLogCommand(options: HookLogOptions = {}): Promise<void> {
  const root = process.cwd();

  // --prune: manual prune and exit
  if (options.prune) {
    const before = await readHookLog(root);
    await readHookLog(root, { prune: true });
    // give async rewrite a chance to settle
    await new Promise((r) => setTimeout(r, 100));
    const after = await readHookLog(root);
    const pruned = before.length - after.length;
    if (pruned > 0) {
      console.log(chalk.dim(`Pruned ${pruned} entr${pruned === 1 ? "y" : "ies"} older than 90 days.`));
    } else {
      console.log(chalk.dim("Nothing to prune."));
    }
    return;
  }

  // Read all entries (with lazy prune)
  const allEntries = await readHookLog(root, { prune: true });

  // --json: raw NDJSON output
  if (options.json) {
    for (const entry of allEntries) {
      process.stdout.write(JSON.stringify(entry) + "\n");
    }
    return;
  }

  if (allEntries.length === 0) {
    console.log("No activity recorded yet. AgentFence hooks will log here automatically.");
    return;
  }

  const daysToShow = Math.max(1, parseInt(options.days ?? "2", 10) || 2);
  const since = new Date();
  since.setDate(since.getDate() - (daysToShow - 1));
  since.setHours(0, 0, 0, 0);

  const recentEntries = allEntries.filter((e) => new Date(e.ts) >= since);

  const todayKey = new Date().toISOString().slice(0, 10);
  const yesterdayDate = new Date();
  yesterdayDate.setDate(yesterdayDate.getDate() - 1);
  const yesterdayKey = yesterdayDate.toISOString().slice(0, 10);

  // 30-day summary (always computed)
  const thirtyDaysAgo = new Date();
  thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30);
  thirtyDaysAgo.setHours(0, 0, 0, 0);
  const last30 = allEntries.filter((e) => new Date(e.ts) >= thirtyDaysAgo);

  // --summary: stats only
  if (options.summary) {
    console.log(chalk.bold("\nLast 30 days"));
    console.log(chalk.dim("─".repeat(61)));
    console.log(renderSummaryLine(last30));
    console.log();
    return;
  }

  // Default: grouped by day
  console.log(chalk.bold("\nAgentFence Activity Log\n"));

  // Group by date
  const byDay = new Map<string, HookLogEntry[]>();
  for (const entry of recentEntries) {
    const key = toDateKey(entry.ts);
    if (!byDay.has(key)) byDay.set(key, []);
    byDay.get(key)!.push(entry);
  }

  const sortedDays = Array.from(byDay.keys()).sort((a, b) => b.localeCompare(a));

  for (const dayKey of sortedDays) {
    const dayEntries = byDay.get(dayKey)!;
    console.log(chalk.bold(dayLabel(dayKey, todayKey, yesterdayKey)));
    console.log(chalk.dim("─".repeat(61)));

    let prevTs: Date | null = null;
    for (const entry of dayEntries) {
      const entryTs = new Date(entry.ts);
      if (prevTs && entryTs.getTime() - prevTs.getTime() > 30 * 60 * 1000) {
        console.log(chalk.dim("  · · ·"));
      }
      prevTs = entryTs;

      const icon = ICONS[entry.outcome];
      const time = formatTime(entry.ts);
      const tool = entry.tool.padEnd(5);
      const fp = displayPath(entry.filePath).padEnd(20);
      console.log(`  ${time}  ${icon}  ${tool}  ${fp}  ${outcomeLabel(entry)}`);
    }
    console.log();
  }

  // 30-day summary footer
  console.log(chalk.bold("Last 30 days"));
  console.log(chalk.dim("─".repeat(61)));
  console.log(renderSummaryLine(last30));
  console.log();
}
```
